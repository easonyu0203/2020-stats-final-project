        var _lastToastr;
        var toastrOptions = {
        		"closeButton": true,

        		"onclick": null,

        function showMsgToastr( type, msg, title )
        {
        	if ( title == undefined ) title = type;
			type = type.toLowerCase();

			if ( _lastToastr) {
				toastr.clear(_lastToastr);
			}

			toastr.options = toastrOptions;
			_lastToastr = toastr[type](msg, title);
        }

        function showMsg( content )
        {
           content = content || '';
           bootbox.alert({
               message: htmlentities(content, 'ENT_QUOTES', 'UTF-8'),
               buttons: { ok: {label : '訊息框-OK', className : 'btn-default'} }
           });
        }

        function showConfirm( content, trueCallback, falseCallback, btnTextConfirm )
        {
        	content = content || '';
        	falseCallback = falseCallback || '';
        	btnTextConfirm = btnTextConfirm || '';
        	btnTextConfirm = (btnTextConfirm !== '') ? btnTextConfirm : '確認訊息框-確認';

        	bootbox.confirm({
                message: content,
                buttons: {
                	confirm: {label : btnTextConfirm, className : 'btn-default'},
                	cancel: {label : '確認訊息框-取消', className : 'btn-default'},
                },
                callback: function(result) {
                    if ( result ) trueCallback();
                    else if ( !result && falseCallback !== '' ) falseCallback();
                }
            });
        }
